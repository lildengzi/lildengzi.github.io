#include<stdio.h>
#include<string.h>
#include<stdlib.h>

int a[10010]{0}, b[10010]{0}, c[10010]{0};
char llLongA[10010], llLongB[10010];

int main()
{
    scanf("%s %s",llLongA, llLongB);
    int lenA = strlen(llLongA), lenB = strlen(llLongB);

    for(int i = 0; i < lenA; i++){a[lenA - i - 1] = llLongA[i] - 48;}
    for(int i = 0; i < lenB; i++){b[lenB - i - 1] = llLongB[i] - 48;}

    //int Len = lenA + LenB - 1; // 乘法最小空间
    int m = lenA > lenB?lenA : lenB; // 加法最小空间

// 加法实现
    for(int i = 0; i < m; i++)
    {
        c[i] = a[i] + b[i];
        if(c[i] >= 10)
        {
            c[i+1]++;
            c[i] -= 10;
        }
    }

    while(c[m] == 0 && m >= 1)m--;

    for(int i = m; i >= 0; i--)
        printf("%d\n", c[i]);
        
    system("pause");
    return 0;
}
